   124→                <Input id="date" type="date" value={date} onChange={(e) => setDate(e.target.value)} className="h-12" />

   129→                  <SelectTrigger className="h-12"><SelectValue placeholder="Pilih kategori" /></SelectTrigger>

   131→                    {Object.entries(groupCategories(categories)).map(([group, cats]) => (

   133→                        <div className="px-2 py-1 text-xs font-semibold text-muted-foreground uppercase">{group}</div>

   135→                          <SelectItem key={c.id} value={String(c.id)} className="py-2.5">
   136→                            <span className="flex items-center justify-between w-full">
   137→                              <span>{c.name}</span>
   138→                              <span className="text-xs text-muted-foreground ml-2">{formatRupiah(c.default_fee)}</span>
   139→                            </span>
   140→                          </SelectItem>
   141→                        ))}
   142→                      </div>
   143→                    ))}
   144→                  </SelectContent>
   145→                </Select>
   146→              </div>
   147→            </div>
   148→
   149→            <div className="grid grid-cols-2 gap-3">
   150→              <div className="space-y-1.5">
   151→                <Label htmlFor="qty" className="text-sm font-medium">Jumlah (qty)</Label>

   154→                  onChange={(e) => setQty(formatNum(e.target.value))}

   156→                />

   158→              </div>

   161→                <RupiahInput

   230→function TransactionList() {
   231→  const [search, setSearch] = useState('')
   232→  const [from, setFrom] = useState('')
   233→  const [to, setTo] = useState('')
   234→  const [minAmount, setMinAmount] = useState('')
   235→  const [maxAmount, setMaxAmount] = useState('')
   236→  const [showAdvanced, setShowAdvanced] = useState(false)
   237→  const [deleteId, setDeleteId] = useState<number | null>(null)
   238→  const [editTxn, setEditTxn] = useState<Txn | null>(null)
   239→  const qc = useQueryClient()
   240→
   241→  const params = new URLSearchParams()
   242→  if (from) params.set('from', from)
   243→  if (to) params.set('to', to)
   244→  if (search) params.set('q', search)
   245→
   246→  const { data, isLoading } = useQuery({
   247→    queryKey: ['transactions', from, to],
   248→    queryFn: () => fetch('/api/transactions?' + params.toString()).then((r) => r.json()),
   249→  })
   250→  const txns: Txn[] = data?.transactions ?? []
   251→
   252→  // Filter berdasarkan teks + rentang nominal (client-side)
   253→  const minVal = parseRupiahInput(minAmount) || 0
   254→  const maxVal = parseRupiahInput(maxAmount) || 0
   255→  const filtered = txns.filter((t) => {
   256→    if (search) {
   257→      const hay = (t.category_name + ' ' + (t.customer_name ?? '') + ' ' + (t.note ?? '')).toLowerCase()
   258→      if (!hay.includes(search.toLowerCase())) return false
   259→    }
   260→    // Amount filter — pakai total (pendapatan bersih) sebagai acuan
   261→    if (minVal > 0 && t.total < minVal) return false
   262→    if (maxVal > 0 && t.total > maxVal) return false
   263→    return true
   264→  })
   265→
   266→  const hasAdvancedFilter = minVal > 0 || maxVal > 0
   267→
   268→  const deleteMutation = useMutation({
   269→    mutationFn: async (id: number) => {
   270→      const res = await fetch(`/api/transactions/${id}`, { method: 'DELETE' })
   271→      if (!res.ok) throw new Error('Gagal hapus')
   272→      return id
   273→    },
   274→    onSuccess: () => {
   275→      toast.success('Transaksi dihapus')
   276→      qc.invalidateQueries({ queryKey: ['transactions'] })
   277→      qc.invalidateQueries({ queryKey: ['summary'] })
   278→      setDeleteId(null)
   279→    },
   280→    onError: () => toast.error('Gagal menghapus transaksi'),
   281→  })
   282→
   283→  const grandAdmin = filtered.reduce((s, t) => s + t.total, 0)
   284→  const grandOmzet = filtered.reduce((s, t) => s + t.total_paid, 0)
   285→  const hasOmzet = filtered.some((t) => t.total_paid > 0)
   286→
   287→  return (
   288→    <Card className="p-5">
   289→      <div className="flex flex-col sm:flex-row gap-2 sm:items-center justify-between mb-4">
   290→        <h2 className="font-bold text-lg">Riwayat Transaksi</h2>
   291→        <div className="flex flex-wrap gap-2">
   292→          <Input type="date" value={from} onChange={(e) => setFrom(e.target.value)} className="h-10 w-auto" />
   293→          <Input type="date" value={to} onChange={(e) => setTo(e.target.value)} className="h-10 w-auto" />
   294→        </div>
   295→      </div>
   296→
   297→      <div className="relative mb-3">
   298→        <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-muted-foreground" />
   299→        <Input value={search} onChange={(e) => setSearch(e.target.value)} placeholder="Cari kategori, pelanggan, atau catatan..." className="pl-9 h-11" />
   300→      </div>
   301→
   302→      {/* Advanced filter toggle */}
   303→      <div className="flex items-center gap-2 mb-3 flex-wrap">
   304→        <button
   305→          onClick={() => setShowAdvanced((v) => !v)}
   306→          className={cn(
   307→            'text-xs px-2.5 py-1.5 rounded-lg border transition-colors flex items-center gap-1.5',
   308→            showAdvanced || hasAdvancedFilter
   309→              ? 'bg-primary/10 border-primary/30 text-primary'
   310→              : 'bg-card hover:bg-secondary text-muted-foreground',
   311→          )}
   312→        >
   313→          <SlidersHorizontal className="w-3.5 h-3.5" />
   314→          Filter Nominal
   315→          {hasAdvancedFilter && <span className="w-1.5 h-1.5 rounded-full bg-primary" />}
   316→        </button>
   317→        {hasAdvancedFilter && (
   318→          <button
   319→            onClick={() => { setMinAmount(''); setMaxAmount('') }}
   320→            className="text-xs px-2.5 py-1.5 rounded-lg text-muted-foreground hover:text-destructive transition-colors"
   321→          >
   322→            Reset
   323→          </button>
   324→        )}
   325→        {(search || from || to || hasAdvancedFilter) && (
   326→          <span className="text-xs text-muted-foreground ml-auto">{filtered.length} dari {txns.length} transaksi</span>
   327→        )}
   328→      </div>
   329→
   330→      {/* Advanced amount range filter */}
   331→      {showAdvanced && (
   332→        <div className="rounded-xl border bg-secondary/30 p-3 mb-3">
   333→          <p className="text-xs font-medium text-muted-foreground mb-2">Rentang Pendapatan Bersih (fee admin)</p>
   334→          <div className="flex items-center gap-2">
   335→            <div className="flex-1">
   336→              <RupiahInput value={minAmount} onChange={setMinAmount} placeholder="Min: 0" className="h-10 text-sm tabular-nums" />
   337→            </div>
   338→            <span className="text-muted-foreground text-sm shrink-0">—</span>
   339→            <div className="flex-1">
   340→              <RupiahInput value={maxAmount} onChange={setMaxAmount} placeholder="Max: ∞" className="h-10 text-sm tabular-nums" />
   341→            </div>
   342→          </div>
   343→          {(minVal > 0 || maxVal > 0) && (
   344→            <p className="text-xs text-primary mt-2">
   345→              Menampilkan transaksi dengan bersih {minVal > 0 ? formatRupiah(minVal) : 'Rp0'} – {maxVal > 0 ? formatRupiah(maxVal) : 'tanpa batas'}
   346→            </p>
   347→          )}
   348→        </div>
   349→      )}
   350→
   351→      {isLoading ? (
   352→        <div className="space-y-2">{Array.from({ length: 5 }).map((_, i) => <Skeleton key={i} className="h-16 w-full" />)}</div>
   353→      ) : filtered.length === 0 ? (
   354→        <div className="py-12 text-center text-muted-foreground">
   355→          <ReceiptText className="w-10 h-10 mx-auto mb-2 opacity-40" />
   356→          <p>Belum ada transaksi pada rentang ini.</p>
   357→        </div>
   358→      ) : (
   359→        <>
   360→          <div className="space-y-3 max-h-[28rem] overflow-y-auto scroll-thin pr-1">
   361→            {(() => {
   362→              // Group by date (transaksi sudah sorted date DESC dari API)
   363→              const groups: { date: string; items: Txn[] }[] = []
   364→              for (const t of filtered) {
   365→                const last = groups[groups.length - 1]
   366→                if (last && last.date === t.date) last.items.push(t)
   367→                else groups.push({ date: t.date, items: [t] })
   368→              }
   369→              const todayStr = todayISO()
   370→              return groups.map((g) => {
   371→                const dayTotal = g.items.reduce((s, t) => s + t.total, 0)
   372→                const isToday = g.date === todayStr
   373→                return (
   374→                  <div key={g.date}>
   375→                    <div className="flex items-center justify-between px-1 mb-1.5 sticky top-0 bg-card/95 backdrop-blur py-1 z-10">
   376→                      <span className="text-xs font-semibold text-muted-foreground flex items-center gap-1.5">
   377→                        {isToday && <span className="w-1.5 h-1.5 rounded-full bg-primary" />}
   378→                        {formatLongDate(g.date)}
   379→                      </span>
   380→                      <span className="text-xs font-semibold tabular-nums text-muted-foreground">{formatRupiah(dayTotal)}</span>
   381→                    </div>
   382→                    <div className="space-y-1.5">
   383→                      {g.items.map((t) => {
   384→                        return (
   385→                          <div key={t.id} className="flex items-center gap-3 p-3 rounded-xl hover:bg-secondary/60 hover:shadow-sm transition-all group border border-transparent hover:border-border/50">
   386→                            <div className={cn('w-11 h-11 rounded-xl flex flex-col items-center justify-center shrink-0 font-bold shadow-sm', getCategoryColor(t.category_group).bg, getCategoryColor(t.category_group).text)}>
   387→                              <span className="text-[10px] leading-none opacity-70">{t.qty}x</span>
   388→                              <span className="text-xs leading-tight">{getCategoryInitial(t.category_name)}</span>
   389→                            </div>
   390→                            <div className="min-w-0 flex-1">
   391→                              <div className="font-medium truncate flex items-center gap-1.5">
   392→                                {t.category_name}
   393→                                {t.customer_name && (
   394→                                  <span className="inline-flex items-center gap-0.5 text-[10px] font-medium px-1.5 py-0.5 rounded-full bg-primary/10 text-primary shrink-0">
   395→                                    <User className="w-2.5 h-2.5" />{t.customer_name}
   396→                                  </span>
   397→                                )}
   398→                              </div>
   399→                              <div className="text-xs text-muted-foreground flex items-center gap-1.5 flex-wrap">
   400→                                <span className="tabular-nums">{t.qty} × {formatRupiah(t.fee_per_unit)}</span>
   401→                                {t.note && <span className="opacity-60">· {t.note}</span>}
   402→                              </div>
   403→                              {t.total_paid > 0 && (
   404→                                <div className="text-[11px] text-muted-foreground/80 mt-0.5 tabular-nums">Omzet {formatRupiah(t.total_paid)}</div>
   405→                              )}
   406→                            </div>
   407→                            <div className="font-bold text-success tabular-nums shrink-0 text-lg">+{formatRupiah(t.total)}</div>
   408→                            <div className="flex items-center gap-0.5 shrink-0">
   409→                              <button
   410→                                onClick={() => setEditTxn(t)}
   411→                                className="opacity-0 group-hover:opacity-100 transition-opacity p-2 text-muted-foreground hover:text-primary hover:bg-primary/10 rounded-lg"
   412→                                aria-label="Edit transaksi"
   413→                              >
   414→                                <Pencil className="w-4 h-4" />
   415→                              </button>
   416→                              <button
   417→                                onClick={() => setDeleteId(t.id)}
   418→                                className="opacity-0 group-hover:opacity-100 transition-opacity p-2 text-muted-foreground hover:text-destructive hover:bg-destructive/10 rounded-lg"
   419→                                aria-label="Hapus transaksi"
   420→                              >
   421→                                <Trash2 className="w-4 h-4" />
   422→                              </button>
   423→                            </div>
   424→                          </div>
   425→                        )
   426→                      })}
   427→                    </div>
   428→                  </div>
   429→                )
   430→              })
   431→            })()}

   434→            <div className="flex items-center justify-between font-semibold">

   443→            )}

   447→
   448→      <AlertDialog open={deleteId !== null} onOpenChange={(v) => !v && setDeleteId(null)}>

   450→          <AlertDialogHeader>

   453→          </AlertDialogHeader>
   454→          <AlertDialogFooter>
   455→            <AlertDialogCancel className="h-11" disabled={deleteMutation.isPending}>Batal</AlertDialogCancel>
   456→            <AlertDialogAction
   457→              onClick={() => deleteId && deleteMutation.mutate(deleteId)}
   458→              disabled={deleteMutation.isPending}
   459→              className="h-11 bg-destructive hover:bg-destructive/90"
   460→            >
   461→              {deleteMutation.isPending ? 'Menghapus...' : 'Hapus'}
   462→            </AlertDialogAction>
   463→          </AlertDialogFooter>
   464→        </AlertDialogContent>
   465→      </AlertDialog>
   466→
   467→      {editTxn && (
   468→        <EditTransactionDialog
   469→          txn={editTxn}
   470→          onClose={() => setEditTxn(null)}
   471→          onSaved={() => {
   472→            qc.invalidateQueries({ queryKey: ['transactions'] })
   473→            qc.invalidateQueries({ queryKey: ['summary'] })
   474→            setEditTxn(null)
   475→          }}
   476→        />
   477→      )}
   478→    </Card>
   479→  )

   482→function EditTransactionDialog({ txn, onClose, onSaved }: { txn: Txn; onClose: () => void; onSaved: () => void }) {
   483→  const [date, setDate] = useState(txn.date)
   484→  const [qty, setQty] = useState(String(txn.qty))
   485→  const [fee, setFee] = useState(String(txn.fee_per_unit))
   486→  const [totalPaid, setTotalPaid] = useState(String(txn.total_paid))
   487→  const [customerName, setCustomerName] = useState(txn.customer_name ?? '')
   488→  const [recordedBy, setRecordedBy] = useState(txn.recorded_by ?? '')
   489→  const [note, setNote] = useState(txn.note ?? '')

   502→          date,

   508→          note: note.trim() || null,
   509→        }),

   575→    </Dialog>
